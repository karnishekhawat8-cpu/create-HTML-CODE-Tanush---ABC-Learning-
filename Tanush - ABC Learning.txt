let index = 0;

function speakText(letter, word) {
  window.speechSynthesis.cancel();
  const speech = new SpeechSynthesisUtterance(`${letter} for ${word}`);
  speech.rate = 0.8;
  speech.pitch = 1.1;
  window.speechSynthesis.speak(speech);
}

function nextLetter() {
  index = (index + 1) % alphabets.length;
  letter.innerText = alphabets[index].l;
  word.innerText = alphabets[index].w;
  image.src = alphabets[index].img;
  speakText(alphabets[index].l, alphabets[index].w);
}
